本文档详细描述了如何为 MerchantConnectFrontend 项目实现多钱包集成方案，以支持 Phantom、OKX 以及未来可能集成的其他钱包。

```typescript
// src/wallets/types.ts

// 钱包连接结果
export interface WalletConnectionResult {
  publicKey: PublicKey;         // 钱包公钥
  connected: boolean;            // 连接状态
  walletName: string;            // 钱包名称
  session?: string;              // 可选会话ID
  autoApprove?: boolean;         // 是否自动批准后续请求
  chainId?: string;              // 链ID
}

// 交易发送选项
export interface SendOptions {
  skipPreflight?: boolean;       // 是否跳过预检
  preflightCommitment?: string;   // 预检确认级别
  maxRetries?: number;           // 最大重试次数
  commitment?: Commitment;       // 确认级别
  // 其他钱包特定选项
}

// 钱包适配器接口
export interface WalletAdapter {
  // 钱包名称
  readonly name: string;
  
  // 钱包图标URL
  readonly icon: string;
  
  // 下载链接
  readonly downloadUrl: {
    android?: string;        // Android应用商店链接
    ios?: string;            // iOS App Store链接
    chrome?: string;         // Chrome扩展商店链接
    firefox?: string;        // Firefox扩展商店链接
  };


  // 是否已安装
  isInstalled(): boolean;

  // 检查是否支持当前环境（浏览器/移动端）
  isSupported(): boolean;

  // 连接钱包
  connect(params?: {
    onlyIfTrusted?: boolean;  // 仅当已授权时连接
    returnURL?: string;       // 移动端返回URL
    appInstallURL?: string;   // 应用安装URL
  }): Promise<WalletConnectionResult>;

  // 断开连接
  disconnect(): Promise<void>;

  // 获取公钥
  getPublicKey(): PublicKey | null;

  // 签名并发送交易
  signAndSendTransaction(
    transaction: Transaction,
    options?: {
      skipPreflight?: boolean;    // 是否跳过预检
      maxRetries?: number;        // 最大重试次数
      commitment?: Commitment;    // 确认级别
      onProgress?: (status: 'pending' | 'confirmed' | 'finalized') => void; // 交易状态回调
    }
  ): Promise<TransactionSignature>;

  // 签名消息
  signMessage(message: Uint8Array): Promise<Uint8Array>;

  // ===== 移动端Deeplink支持 =====
  
  // 是否支持Deeplink
  supportsDeeplink: boolean;
  
  // 生成Deeplink URL
  generateDeeplink(params: {
    action: 'connect' | 'signTransaction' | 'signMessage';
    params: Record<string, any>;
    returnURL: string;
  }): string;
  
  // 处理Deeplink返回
  handleDeeplink(url: URL): Promise<void>;

  // ===== 事件 =====
  on(event: 'connect', callback: (publicKey: PublicKey) => void): void;
  on(event: 'disconnect', callback: () => void): void;
  on(event: 'error', callback: (error: WalletError) => void): void;
  on(event: 'ready', callback: () => void): void;
  off(event: string, callback: (...args: any[]) => void): void;
}

// 钱包错误类型
export class WalletError extends Error {
  code: number;
  message: string;
  
  constructor(code: number, message: string) {
    super(message);
    this.code = code;
    this.name = 'WalletError';
  }
}
```

```typescript
// src/wallets/phantom/PhantomWalletAdapter.ts
import { WalletAdapter, WalletConnectionResult, WalletError } from '../types';
import { PublicKey, Transaction, SendOptions, Commitment } from '@solana/web3.js';

declare global {
  interface Window {
    phantom?: {
      solana: any;
    };
  }
}

export class PhantomWalletAdapter implements WalletAdapter {
  name = 'Phantom';
  icon = 'https://phantom.app/favicon.ico';
  supportsDeeplink = true;
  
  downloadUrl = {
    android: 'https://play.google.com/store/apps/details?id=app.phantom',
    ios: 'https://apps.apple.com/us/app/phantom-crypto-wallet/id1598432977',
    chrome: 'https://chrome.google.com/webstore/detail/phantom/bfnaelmomeimhlpmgjnjophhpkkoljpa',
  };

  private _publicKey: PublicKey | null = null;
  private _connected = false;
  private _listeners: Map<string, Set<Function>> = new Map();
  private _provider: any;
  private _deeplinkHandlers: Map<string, (params: any) => void> = new Map();

  constructor() {
    this._setupDeeplinkHandlers();
  }

  // 检查是否已安装
  isInstalled(): boolean {
    return !!window.phantom?.solana?.isPhantom;
  }


  // 检查是否支持当前环境
  isSupported(): boolean {
    return this.isInstalled() || this._isMobile();
  }

  // 连接钱包
  async connect(params?: {
    onlyIfTrusted?: boolean;
    returnURL?: string;
    appInstallURL?: string;
  }): Promise<WalletConnectionResult> {
    if (this._isMobile() && !this.isInstalled()) {
      // 移动端未安装，使用Deeplink
      const deeplink = this.generateDeeplink({
        action: 'connect',
        params: { onlyIfTrusted: params?.onlyIfTrusted },
        returnURL: params?.returnURL || window.location.href
      });
      
      if (params?.appInstallURL) {
        // 检查是否安装了应用
        const isInstalled = await this._checkAppInstalled(params.appInstallURL);
        if (!isInstalled) {
          window.open(params.appInstallURL, '_blank');
          throw new WalletError(4001, 'Phantom app not installed');
        }
      }
      
      window.location.href = deeplink;
      return new Promise((_, reject) => {
        // 设置超时
        setTimeout(() => {
          reject(new WalletError(4001, 'User rejected the request'));
        }, 30000); // 30秒超时
      });
    }

    // 桌面端或已安装移动端使用标准流程
    try {
      this._provider = window.phantom?.solana;
      if (!this._provider) {
        throw new WalletError(4001, 'Phantom not installed');
      }

      const response = await this._provider.connect({ onlyIfTrusted: params?.onlyIfTrusted });
      this._publicKey = new PublicKey(response.publicKey);
      this._connected = true;
      
      this._emit('connect', this._publicKey);
      
      return {
        publicKey: this._publicKey,
        connected: true,
        walletName: this.name,
        autoApprove: true
      };
    } catch (error: any) {
      throw new WalletError(error.code || 4000, error.message);
    }
  }

  // 生成Deeplink URL
  generateDeeplink(params: {
    action: 'connect' | 'signTransaction' | 'signMessage';
    params: Record<string, any>;
    returnURL: string;
  }): string {
    const { action, params: actionParams, returnURL } = params;
    const baseUrl = `https://phantom.app/ul/v1/${action}`;
    const query = new URLSearchParams({
      ...actionParams,
      redirectLink: returnURL,
      app_url: window.location.href,
      ref: 'your_app_name'
    });
    
    return `${baseUrl}?${query.toString()}`;
  }

  // 处理Deeplink返回
  async handleDeeplink(url: URL): Promise<void> {
    const params = new URLSearchParams(url.search);
    const action = url.pathname.split('/').pop();
    const handler = this._deeplinkHandlers.get(action);
    
    if (handler) {
      const result = Object.fromEntries(params.entries());
      handler(result);
    }
  }

  // 签名并发送交易（支持Deeplink）
  async signAndSendTransaction(
    transaction: Transaction,
    options?: {
      skipPreflight?: boolean;
      maxRetries?: number;
      commitment?: Commitment;
      onProgress?: (status: string) => void;
    }
  ): Promise<TransactionSignature> {
    if (this._isMobile() && !this.isInstalled()) {
      // 移动端未安装，使用Deeplink
      const serializedTx = transaction.serialize({ requireAllSignatures: false }).toString('base64');
      const deeplink = this.generateDeeplink({
        action: 'signTransaction',
        params: {
          transaction: serializedTx,
          display: 'touch',
          chain: 'solana:mainnet',
          ...options
        },
        returnURL: window.location.href
      });

      window.location.href = deeplink;
      
      return new Promise((resolve, reject) => {
        const handler = (result: any) => {
          if (result.signature) {
            resolve(result.signature);
          } else {
            reject(new WalletError(4001, result.error || 'User rejected the request'));
          }
          this._deeplinkHandlers.delete('signTransaction');
        };
        
        this._deeplinkHandlers.set('signTransaction', handler);
        
        // 设置超时
        setTimeout(() => {
          this._deeplinkHandlers.delete('signTransaction');
          reject(new WalletError(4001, 'Request timeout'));
        }, 30000); // 30秒超时
      });
    }

    // 桌面端或已安装移动端使用标准流程
    try {
      if (!this._provider) {
        throw new WalletError(4001, 'Wallet not connected');
      }
      
      const { signature } = await this._provider.signAndSendTransaction(
        transaction,
        options
      );
      
      return signature;
    } catch (error: any) {
      throw new WalletError(error.code || 4000, error.message);
    }
  }

  // 私有方法
  private _setupDeeplinkHandlers() {
    // 处理从Deeplink返回的连接响应
    this._deeplinkHandlers.set('connect', (params: any) => {
      if (params.publicKey) {
        this._publicKey = new PublicKey(params.publicKey);
        this._connected = true;
        this._emit('connect', this._publicKey);
      } else if (params.error) {
        this._emit('error', new WalletError(4001, params.error));
      }
    });
  }

  private _isMobile(): boolean {
    return /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent);
  }

  private async _checkAppInstalled(appUrl: string): Promise<boolean> {
    return new Promise((resolve) => {
      const iframe = document.createElement('iframe');
      iframe.src = appUrl;
      iframe.style.display = 'none';
      
      const timeout = setTimeout(() => {
        document.body.removeChild(iframe);
        resolve(false);
      }, 2000);
      
      iframe.onload = () => {
        clearTimeout(timeout);
        document.body.removeChild(iframe);
        resolve(true);
      };
      
      document.body.appendChild(iframe);
    });
  }

  private _emit(event: string, ...args: any[]) {
    const listeners = this._listeners.get(event);
    if (listeners) {
      listeners.forEach(listener => listener(...args));
    }
  }

  // 实现其他必要的方法...
}
```

## 4. 使用示例

### 4.1 初始化钱包适配器

```typescript
// src/hooks/useWallet.ts
import { useState, useEffect, useCallback } from 'react';
import { WalletFactory } from '../wallets/WalletFactory';
import { WalletType } from '../wallets/types';

export function useWallet() {
  const [wallet, setWallet] = useState<any>(null);
  const [publicKey, setPublicKey] = useState<string | null>(null);
  const [connected, setConnected] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  // 初始化钱包
  useEffect(() => {
    const initWallet = async () => {
      try {
        // 获取Phantom钱包实例
        const phantom = WalletFactory.getWallet(WalletType.Phantom, {
          returnURL: window.location.href,
          appInstallURL: 'https://phantom.app/download'
        });

        if (!phantom) {
          throw new Error('Phantom wallet is not supported in this browser');
        }

        // 设置事件监听
        phantom.on('connect', (pk: any) => {
          setPublicKey(pk.toString());
          setConnected(true);
          setError(null);
        });

        phantom.on('disconnect', () => {
          setPublicKey(null);
          setConnected(false);
        });

        phantom.on('error', (err: Error) => {
          setError(err);
          console.error('Wallet error:', err);
        });

        setWallet(phantom);

        // 尝试静默连接
        if (phantom.isInstalled()) {
          await phantom.connect({ onlyIfTrusted: true });
        }

        // 处理Deeplink返回
        const url = new URL(window.location.href);
        if (url.searchParams.has('publicKey') || url.searchParams.has('error')) {
          await phantom.handleDeeplink(url);
          // 清除URL参数
          window.history.replaceState({}, document.title, window.location.pathname);
        }
      } catch (err) {
        console.error('Failed to initialize wallet:', err);
        setError(err instanceof Error ? err : new Error('Unknown error'));
      }
    };

    initWallet();

    return () => {
      // 清理
      if (wallet) {
        wallet.off('connect');
        wallet.off('disconnect');
        wallet.off('error');
      }
    };
  }, []);

  // 连接钱包
  const connect = useCallback(async () => {
    if (!wallet) return;
    
    try {
      await wallet.connect({
        onlyIfTrusted: false,
        returnURL: window.location.href,
        appInstallURL: 'https://phantom.app/download'
      });
    } catch (err) {
      console.error('Failed to connect wallet:', err);
      throw err;
    }
  }, [wallet]);

  // 断开连接
  const disconnect = useCallback(async () => {
    if (wallet && connected) {
      await wallet.disconnect();
    }
  }, [wallet, connected]);

  // 发送交易
  const sendTransaction = useCallback(async (transaction: any) => {
    if (!wallet || !connected) {
      throw new Error('Wallet not connected');
    }

    try {
      const signature = await wallet.signAndSendTransaction(transaction, {
        skipPreflight: false,
        onProgress: (status: string) => {
          console.log('Transaction status:', status);
        }
      });
      return signature;
    } catch (err) {
      console.error('Transaction failed:', err);
      throw err;
    }
  }, [wallet, connected]);

  return {
    wallet,
    publicKey,
    connected,
    error,
    connect,
    disconnect,
    sendTransaction,
  };
}
```

### 4.3 交易超时

处理方案：

设置交易超时时间
提供重试机制

```typescript
async function sendWithRetry(
wallet: WalletAdapter,
transaction: Transaction,
options: SendOptions = {},
maxRetries = 3
): Promise<string> {
let lastError;

for (let i = 0; i < maxRetries; i++) {
try {
return await wallet.signAndSendTransaction(transaction, {
...options,
maxRetries: 3,
});
} catch (error) {
lastError = error;
console.warn(`Attempt ${i + 1} failed:`, error);
await new Promise(resolve => setTimeout(resolve, 1000 \* (i + 1))); // 指数退避
}
}

throw lastError || new Error('Transaction failed after multiple attempts');
}
```

## 5. 实现细节

### 5.1 钱包连接流程

#### 5.1.1 初始化阶段
1. 生成DApp密钥对
   - 使用`nacl.box.keyPair()`生成Ed25519密钥对
   - 将密钥对Base58编码后存储在localStorage中
   - 密钥对用于后续的端到端加密通信

2. 检查已保存的钱包会话
   - 从localStorage加载`phantom_encryption_public_key`
   - 加载已保存的`phantom_public_key`和`phantom_session`
   - 如果存在有效会话，自动设置为已连接状态

#### 5.1.2 连接流程
1. 生成Deeplink URL
   - 使用`openPhantomConnectDeeplink`创建连接Deeplink
   - 包含DApp公钥、重定向URL等参数
   - 支持Universal Links和自定义URL Scheme

2. 处理连接回调
   - 解析URL参数中的`phantom_encryption_public_key`、`nonce`和`data`
   - 使用DApp私钥和Phantom公钥计算共享密钥
   - 解密连接响应数据，获取钱包公钥和会话令牌
   - 将会话信息保存到localStorage

### 5.2 交易处理流程

#### 5.2.1 交易创建
1. 构建交易
   - 使用`@solana/web3.js`创建交易对象
   - 设置`feePayer`为当前连接的钱包地址
   - 添加必要的交易指令

2. 估算交易费用
   - 调用`estimateTransactionFee`估算网络费用
   - 显示预估费用给用户确认
   - 处理费用估算失败的情况

#### 5.2.2 交易签名与发送
1. 准备交易数据
   - 序列化交易对象
   - 构建交易负载，包含会话信息
   - 使用共享密钥加密交易数据

2. 发送交易
   - 生成交易Deeplink
   - 在移动端使用`window.location.href`跳转
   - 在桌面端使用`<a>`标签模拟点击

#### 5.2.3 交易响应处理
1. 解析交易响应
   - 处理URL中的`data`和`nonce`参数
   - 使用共享密钥解密响应数据
   - 提取交易签名

2. 错误处理
   - 处理用户拒绝交易的情况
   - 处理网络错误和超时
   - 提供友好的错误提示

### 5.3 安全性考虑

#### 5.3.1 密钥管理
- DApp密钥对存储在浏览器的localStorage中
- 私钥永远不会发送到服务器
- 建议定期轮换密钥对

#### 5.3.2 通信安全
- 所有敏感数据使用`nacl`进行端到端加密
- 使用随机nonce防止重放攻击
- 验证交易签名确保数据完整性

#### 5.3.3 会话管理
- 会话令牌具有时效性
- 断开连接时清除本地存储的会话信息
- 支持手动断开连接

### 5.4 性能优化

#### 5.4.1 交易缓存
- 使用sessionStorage缓存待处理的交易
- 页面刷新后可以恢复交易状态
- 交易完成后自动清理缓存

#### 5.4.2 懒加载
- 按需加载钱包适配器
- 减少初始加载时间
- 优化移动端性能

### 5.5 兼容性处理

#### 5.5.1 多平台支持
- 检测用户设备类型
- 根据平台选择合适的Deeplink打开方式
- 提供备选方案处理不兼容情况

#### 5.5.2 浏览器兼容性
- 支持主流现代浏览器
- 优雅降级处理不支持的API
- 提供明确的错误提示
