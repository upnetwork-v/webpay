```typescript
// src/wallets/types.ts
export interface WalletAdapter {
// 钱包名称
readonly name: string;

// 是否已安装
isInstalled(): boolean;

// 连接钱包
connect(): Promise<WalletConnectionResult>;

// 断开连接
disconnect(): Promise<void>;

// 获取公钥
getPublicKey(): PublicKey | null;

// 签名并发送交易
signAndSendTransaction(
transaction: Transaction,
options?: SendOptions
): Promise<TransactionSignature>;

// 签名消息
signMessage(message: Uint8Array): Promise<Uint8Array>;

// 事件监听
on(event: 'connect' | 'disconnect' | 'error', callback: (arg: any) => void): void;
off(event: string, callback: (arg: any) => void): void;
}

export interface WalletConnectionResult {
publicKey: PublicKey;
connected: boolean;
chainId?: string; // 支持多链
}

export interface SendOptions {
skipPreflight?: boolean;
preflightCommitment?: string;
maxRetries?: number;
```

```typescript
    // 显示安装引导

    return;
  }
  // 连接逻辑...

```

#### 4.2 网络不匹配

处理方案：

检测当前网络是否与所需网络匹配
提供切换网络的引导

```typescript
async function ensureCorrectNetwork(wallet: WalletAdapter) {
  const currentNetwork = await wallet.getNetwork();
  if (currentNetwork !== TARGET_NETWORK) {
    throw new Error(`请将钱包切换到 ${TARGET_NETWORK} 网络`);
  }
}
```

#### 4.3 交易超时

处理方案：

设置交易超时时间
提供重试机制

```typescript
async function sendWithRetry(
wallet: WalletAdapter,
transaction: Transaction,
options: SendOptions = {},
maxRetries = 3
): Promise<string> {
let lastError;

for (let i = 0; i < maxRetries; i++) {
try {
return await wallet.signAndSendTransaction(transaction, {
...options,
maxRetries: 3,
});
} catch (error) {
lastError = error;
console.warn(`Attempt ${i + 1} failed:`, error);
await new Promise(resolve => setTimeout(resolve, 1000 \* (i + 1))); // 指数退避
}
}

throw lastError || new Error('Transaction failed after multiple attempts');
}
```

#### 4.4 用户拒绝连接/交易

处理方案：

捕获用户拒绝错误
提供友好的错误提示

```typescript
try {

} catch (error) {
  if (isUserRejectedError(error)) {

  }

}
```

```typescript
// src/utils/walletUtils.ts
export function detectWallet(walletName: string): boolean {
  switch (walletName.toLowerCase()) {
    case "phantom":
      return !!(window as any).phantom?.solana?.isPhantom;
    case "okx":
      return !!(window as any).okxwallet?.solana;
    default:
      return false;
  }
}

export function getWalletInstallLink(walletName: string): string {
  const links = {
    phantom: "https://phantom.app/download",
    okx: "https://www.okx.com/web3",
    // 添加更多钱包...
  };

  return links[walletName.toLowerCase()] || "#";
}
```

```typescript
// src/utils/errorUtils.ts
export function isUserRejectedError(error: any): boolean {
  // Phantom 错误
  if (error.message?.includes("User rejected the request")) {
    return true;
  }

  // OKX 错误
  if (error.code === 4001) {
    return true;
  }

  return false;
}

export function getErrorMessage(error: any): string {
  if (isUserRejectedError(error)) {
    return "您已取消操作";
  }

  // 其他错误处理...

  return error.message || "发生未知错误";
}
```
